import 'package:flutter/material.dart';
import 'package:flutter/services.dart';
import 'package:shared_preferences/shared_preferences.dart';

class GeneratePage extends StatefulWidget {
  const GeneratePage({super.key});

  @override
  State<GeneratePage> createState() => _GeneratePageState();
}

class _GeneratePageState extends State<GeneratePage> {
  static const String _varietiesKey = 'varieties';
  static const String _itemNamesKey = 'item_names';
  static const String _lessCategoriesKey = 'less_categories';
  String? _selectedCategory;
  String? _selectedGender;
  String? _selectedVariety;
  String? _selectedItemName;
  String? _selectedMakingType;
  String? _selectedReturnPercent;
  String? _selectedHallmarkType;
  String? _selectedHallmarkValue;

  bool _searchable = false;
  bool _crud = false;
  bool _persistence = false;

  final List<String> categories = ['Gold22kt', 'Gold18kt', 'Silver'];
  final List<String> genders = ['Gents', 'Ladies', 'Common'];
  final List<String> _defaultVarieties = [
    'German',
    'Italian',
    'Turkish',
    'Yellow',
  ];
  final List<String> _varieties = [];
  final List<String> _defaultItemNames = [
    'Chain',
    'Bracelets',
    'Mangal Sutra',
    'Pendals',
  ];
  final List<String> _itemNames = [];
  final List<String> _makingTypes = [
    'FixRate',
    'PerGram',
    'Percentage',
    'TotalMaking',
  ];
  final List<String> _returnPercentOptions = [
    'Fine',
    '92%',
    '90%',
    '80%',
    '85%',
    '70%',
    '75%',
    '60%',
    '65%',
    '50%',
  ];
  final List<String> _hallmarkTypes = [
    'HUID',
    'Hallmarking',
  ];
  final List<String> _hallmarkOptions = [
    '24kt BIS 999',
    '22kt BIS 916',
    '18kt BIS 750',
    '14kt BIS 583',
    '12kt BIS 500',
    '10kt BIS 417',
    '9kt BIS 375',
  ];
  final List<String> _defaultLessCategories = [
    'Stones',
    'Meena',
    'Kundan',
  ];
  final List<String> _lessCategories = [];
  final TextEditingController _makingChargeController =
      TextEditingController();
  final TextEditingController _huidController = TextEditingController();
  final TextEditingController _varietyController = TextEditingController();
  final TextEditingController _itemNameController = TextEditingController();
  final TextEditingController _grossWeightController = TextEditingController();
  final TextEditingController _lessWeightController = TextEditingController();
  final TextEditingController _netWeightController = TextEditingController();
  final List<_LessCategoryEntry> _lessCategoryEntries = [
    _LessCategoryEntry(),
  ];

  @override
  void initState() {
    super.initState();
    _grossWeightController.addListener(_updateWeights);
    for (final entry in _lessCategoryEntries) {
      entry.valueController.addListener(_updateWeights);
    }
    _loadPrefs();
  }

  @override
  void dispose() {
    _makingChargeController.dispose();
    _huidController.dispose();
    _varietyController.dispose();
    _itemNameController.dispose();
    _grossWeightController.dispose();
    _lessWeightController.dispose();
    _netWeightController.dispose();
    for (final entry in _lessCategoryEntries) {
      entry.dispose();
    }
    super.dispose();
  }

  Future<void> _loadPrefs() async {
    final prefs = await SharedPreferences.getInstance();
    final persist = prefs.getBool('persistence') ?? false;
    setState(() {
      _varieties
        ..clear()
        ..addAll(
          prefs.getStringList(_varietiesKey) ?? _defaultVarieties,
        );
      _itemNames
        ..clear()
        ..addAll(
          prefs.getStringList(_itemNamesKey) ?? _defaultItemNames,
        );
      _lessCategories
        ..clear()
        ..addAll(
          prefs.getStringList(_lessCategoriesKey) ?? _defaultLessCategories,
        );
      if (persist) {
        _persistence = true;
        _selectedCategory = prefs.getString('category');
        _selectedGender = prefs.getString('gender');
        _selectedVariety = prefs.getString('variety');
        _selectedItemName = prefs.getString('item_name');
        _varietyController.text = _selectedVariety ?? '';
        _itemNameController.text = _selectedItemName ?? '';
        _selectedMakingType = prefs.getString('making_type');
        _selectedReturnPercent = prefs.getString('return_percent');
        _selectedHallmarkType = prefs.getString('hallmark_type');
        _selectedHallmarkValue = prefs.getString('hallmark_value');
        _huidController.text = prefs.getString('huid_value') ?? '';
        _makingChargeController.text =
            prefs.getString('making_charge') ?? '';
        _searchable = prefs.getBool('searchable') ?? false;
        _crud = prefs.getBool('crud') ?? false;
      }
    });
  }

  void _updateWeights() {
    double parse(String v) => double.tryParse(v.trim()) ?? 0.0;
    final gross = parse(_grossWeightController.text);
    double lessTotal = 0;
    for (final entry in _lessCategoryEntries) {
      lessTotal += parse(entry.valueController.text);
    }
    final net = gross - lessTotal;
    _lessWeightController.text = lessTotal.toStringAsFixed(3);
    _netWeightController.text = net.toStringAsFixed(3);
  }

  Future<void> _saveIfNeeded() async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.setStringList(_varietiesKey, _varieties);
    await prefs.setStringList(_itemNamesKey, _itemNames);
    await prefs.setStringList(_lessCategoriesKey, _lessCategories);
    if (_persistence) {
      if (_selectedCategory != null) {
        await prefs.setString('category', _selectedCategory!);
      } else {
        await prefs.remove('category');
      }
      if (_selectedGender != null) {
        await prefs.setString('gender', _selectedGender!);
      } else {
        await prefs.remove('gender');
      }
      if (_selectedVariety != null) {
        await prefs.setString('variety', _selectedVariety!);
      } else {
        await prefs.remove('variety');
      }
      if (_selectedItemName != null) {
        await prefs.setString('item_name', _selectedItemName!);
      } else {
        await prefs.remove('item_name');
      }
      if (_selectedMakingType != null) {
        await prefs.setString('making_type', _selectedMakingType!);
      } else {
        await prefs.remove('making_type');
      }
      if (_selectedReturnPercent != null) {
        await prefs.setString('return_percent', _selectedReturnPercent!);
      } else {
        await prefs.remove('return_percent');
      }
      if (_selectedHallmarkType != null) {
        await prefs.setString('hallmark_type', _selectedHallmarkType!);
      } else {
        await prefs.remove('hallmark_type');
      }
      if (_selectedHallmarkValue != null) {
        await prefs.setString('hallmark_value', _selectedHallmarkValue!);
      } else {
        await prefs.remove('hallmark_value');
      }
      await prefs.setString('huid_value', _huidController.text);
      await prefs.setString('making_charge', _makingChargeController.text);
      await prefs.setBool('searchable', _searchable);
      await prefs.setBool('crud', _crud);
      await prefs.setBool('persistence', _persistence);
    } else {
      await prefs.remove('category');
      await prefs.remove('gender');
      await prefs.remove('variety');
      await prefs.remove('item_name');
      await prefs.remove('making_type');
      await prefs.remove('return_percent');
      await prefs.remove('hallmark_type');
      await prefs.remove('hallmark_value');
      await prefs.remove('huid_value');
      await prefs.remove('making_charge');
      await prefs.remove('searchable');
      await prefs.remove('crud');
      await prefs.setBool('persistence', false);
    }
  }

  Future<void> _openVarietySearch() async {
    FocusScope.of(context).unfocus();
    final selected = await showModalBottomSheet<String>(
      context: context,
      isScrollControlled: true,
      builder: (context) {
        String query = '';
        final controller = TextEditingController();
        return StatefulBuilder(
          builder: (context, setModalState) {
            final filtered = _varieties
                .where(
                  (v) => v.toLowerCase().contains(query.toLowerCase()),
                )
                .toList();
            return GestureDetector(
              behavior: HitTestBehavior.opaque,
              onTap: () => FocusScope.of(context).unfocus(),
              child: SafeArea(
                child: Padding(
                  padding: EdgeInsets.only(
                    left: 16,
                    right: 16,
                    top: 12,
                    bottom: MediaQuery.of(context).viewInsets.bottom + 16,
                  ),
                  child: Column(
                    mainAxisSize: MainAxisSize.min,
                    children: [
                      Row(
                        children: [
                          Expanded(
                            child: TextField(
                              controller: controller,
                              decoration: InputDecoration(
                                labelText: 'Search variety',
                                prefixIcon: const Icon(Icons.search),
                                border: OutlineInputBorder(
                                  borderRadius: BorderRadius.circular(8),
                                ),
                              ),
                              onChanged: (value) {
                                setModalState(() {
                                  query = value;
                                });
                              },
                            ),
                          ),
                          const SizedBox(width: 8),
                          IconButton(
                            tooltip: 'Add variety',
                            icon: const Icon(Icons.add),
                            onPressed: () => _addVariety(setModalState),
                          ),
                        ],
                      ),
                      const SizedBox(height: 12),
                      if (filtered.isEmpty)
                        const Padding(
                          padding: EdgeInsets.symmetric(vertical: 24),
                          child: Text('No matches'),
                        )
                      else
                        ConstrainedBox(
                          constraints: const BoxConstraints(maxHeight: 320),
                          child: ListView.separated(
                            shrinkWrap: true,
                            itemCount: filtered.length,
                            separatorBuilder: (_, _) =>
                                const Divider(height: 1),
                            itemBuilder: (context, index) {
                              final variety = filtered[index];
                              final isSelected = variety == _selectedVariety;
                              return ListTile(
                                title: Text(variety),
                                trailing: Row(
                                  mainAxisSize: MainAxisSize.min,
                                  children: [
                                    if (isSelected)
                                      const Padding(
                                        padding: EdgeInsets.only(right: 4),
                                        child: Icon(Icons.check),
                                      ),
                                    PopupMenuButton<String>(
                                      onSelected: (value) {
                                        if (value == 'edit') {
                                          _editVariety(
                                            variety,
                                            setModalState,
                                          );
                                        } else if (value == 'delete') {
                                          _deleteVariety(
                                            variety,
                                            setModalState,
                                          );
                                        }
                                      },
                                      itemBuilder: (_) => const [
                                        PopupMenuItem(
                                          value: 'edit',
                                          child: Text('Edit'),
                                        ),
                                        PopupMenuItem(
                                          value: 'delete',
                                          child: Text('Delete'),
                                        ),
                                      ],
                                    ),
                                  ],
                                ),
                                onTap: () =>
                                    Navigator.of(context).pop(variety),
                              );
                            },
                          ),
                        ),
                    ],
                  ),
                ),
              ),
            );
          },
        );
      },
    );

    if (selected != null && selected != _selectedVariety) {
      setState(() {
        _selectedVariety = selected;
        _varietyController.text = selected;
      });
      _saveIfNeeded();
    }
  }

  Future<void> _editVariety(
    String current,
    StateSetter setModalState,
  ) async {
    final controller = TextEditingController(text: current);
    final result = await showDialog<String>(
      context: context,
      builder: (context) {
        return AlertDialog(
          title: const Text('Edit variety'),
          content: TextField(
            controller: controller,
            autofocus: true,
            decoration: const InputDecoration(
              labelText: 'Variety',
            ),
          ),
          actions: [
            TextButton(
              onPressed: () => Navigator.of(context).pop(),
              child: const Text('Cancel'),
            ),
            ElevatedButton(
              onPressed: () {
                Navigator.of(context).pop(controller.text);
              },
              child: const Text('Save'),
            ),
          ],
        );
      },
    );

    if (!mounted) {
      return;
    }

    final updated = result?.trim();
    if (updated == null || updated.isEmpty || updated == current) {
      return;
    }
    final duplicate = _varieties.any(
      (v) => v.toLowerCase() == updated.toLowerCase() && v != current,
    );
    if (duplicate) {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(
          content: Text('Variety already exists'),
        ),
      );
      return;
    }

    setState(() {
      final index = _varieties.indexOf(current);
      if (index != -1) {
        _varieties[index] = updated;
      }
      if (_selectedVariety == current) {
        _selectedVariety = updated;
        _varietyController.text = updated;
      }
    });
    setModalState(() {});
    _saveIfNeeded();
  }

    final controller = TextEditingController();

  Future<void> _deleteVariety(

        return AlertDialog(

              child: const Text('Cancel'),

                            itemCount: filtered.length,
                            separatorBuilder: (_, _) =>
                                const Divider(height: 1),
                            itemBuilder: (context, i) {
                              final item = filtered[i];
                              final isSelected =
                                  item == _lessCategoryEntries[index].category;
                              return ListTile(
                                title: Text(item),
                                trailing: Row(
                                  mainAxisSize: MainAxisSize.min,
                                  children: [
                                    if (isSelected)
                                      const Padding(
                                        padding: EdgeInsets.only(right: 4),
                                        child: Icon(Icons.check),
                                      ),
                                    PopupMenuButton<String>(
                                      onSelected: (value) {
                                        if (value == 'edit') {
                                          _editLessCategory(
                                            item,
                                            setModalState,
                                          );
                                        } else if (value == 'delete') {
                                          _deleteLessCategory(
                                            item,
                                            setModalState,
                                          );
                                        }
                                      },
                                      itemBuilder: (_) => const [
                                        PopupMenuItem(
                                          value: 'edit',
                                          child: Text('Edit'),
                                        ),
                                        PopupMenuItem(
                                          value: 'delete',
                                          child: Text('Delete'),
                                        ),
                                      ],
                                    ),
                                  ],
                                ),
                                onTap: () => Navigator.of(context).pop(item),
                              );
                            },
                          ),
                        ),
                    ],
                  ),
                ),
              ),
            );
          },
        );
      },
    );

    if (selected != null) {
      setState(() {
        _lessCategoryEntries[index].category = selected;
        _lessCategoryEntries[index].categoryController.text = selected;
      });
    }
  }

  Future<void> _editLessCategory(
    String current,
    StateSetter setModalState,
  ) async {
    final controller = TextEditingController(text: current);
    final result = await showDialog<String>(
      context: context,
      builder: (context) {
        return AlertDialog(
          title: const Text('Edit category'),
          content: TextField(
            controller: controller,
            autofocus: true,
            decoration: const InputDecoration(
              labelText: 'Category',
            ),
          ),
          actions: [
            TextButton(
              onPressed: () => Navigator.of(context).pop(),
              child: const Text('Cancel'),
            ),
            ElevatedButton(
              onPressed: () {
                Navigator.of(context).pop(controller.text);
              },
              child: const Text('Save'),
            ),
          ],
        );
      },
    );

    if (!mounted) {
      return;
    }

    final updated = result?.trim();
    if (updated == null || updated.isEmpty || updated == current) {
      return;
    }
    final duplicate = _lessCategories.any(
      (v) => v.toLowerCase() == updated.toLowerCase() && v != current,
    );
    if (duplicate) {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(
          content: Text('Category already exists'),
        ),
      );
      return;
    }

    setState(() {
      final index = _lessCategories.indexOf(current);
      if (index != -1) {
        _lessCategories[index] = updated;
      }
      for (final entry in _lessCategoryEntries) {
        if (entry.category == current) {
          entry.category = updated;
          entry.categoryController.text = updated;
        }
      }
    });
    setModalState(() {});
    _saveIfNeeded();
  }

  Future<void> _addLessCategory(StateSetter setModalState) async {
    final controller = TextEditingController();
    final result = await showDialog<String>(
      context: context,
      builder: (context) {
        return AlertDialog(
          title: const Text('Add category'),
          content: TextField(
            controller: controller,
            autofocus: true,
            decoration: const InputDecoration(
              labelText: 'Category',
            ),
          ),
          actions: [
            TextButton(
              onPressed: () => Navigator.of(context).pop(),
              child: const Text('Cancel'),
            ),
            ElevatedButton(
              onPressed: () {
                Navigator.of(context).pop(controller.text);
              },
              child: const Text('Add'),
            ),
          ],
        );
      },
    );

    if (!mounted) {
      return;
    }

    final updated = result?.trim();
    if (updated == null || updated.isEmpty) {
      return;
    }
    final duplicate =
        _lessCategories.any((v) => v.toLowerCase() == updated.toLowerCase());
    if (duplicate) {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(
          content: Text('Category already exists'),
        ),
      );
      return;
    }

    setState(() {
      _lessCategories.add(updated);
    });
    setModalState(() {});
    _saveIfNeeded();
  }

  Future<void> _deleteLessCategory(
    String item,
    StateSetter setModalState,
  ) async {
    final shouldDelete = await showDialog<bool>(
      context: context,
      builder: (context) {
        return AlertDialog(
          title: const Text('Delete category'),
          content: Text('Delete "$item"?'),
          actions: [
            TextButton(
              onPressed: () => Navigator.of(context).pop(false),
              child: const Text('Cancel'),
            ),
            ElevatedButton(
              onPressed: () => Navigator.of(context).pop(true),
              style: ElevatedButton.styleFrom(
                backgroundColor: Colors.red,
                foregroundColor: Colors.white,
              ),
              child: const Text('Delete'),
            ),
          ],
        );
      },
    );

    if (!mounted) {
      return;
    }

    if (shouldDelete != true) {
      return;
    }

    setState(() {
      _lessCategories.remove(item);
      for (final entry in _lessCategoryEntries) {
        if (entry.category == item) {
          entry.category = null;
          entry.categoryController.text = '';
        }
      }
    });
    setModalState(() {});
    _saveIfNeeded();
  }
